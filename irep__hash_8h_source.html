106 template<>
107 inline std::size_t basic_hash_combine<64>(
108  std::size_t h1,
109  std::size_t h2)
110 {
111  #ifdef _MSC_VER
112  // The below intentionally limits the hash key to 32 bits.
113  // This is because Visual Studio's STL masks away anything
114  // but the least significant n bits, where 2^n is the size of
115  // the hash table. On systems with 64-bit size_t, we then see
116  // performance degradation as too much of the hash key is
117  // contained in bits that will be masked away. There is no such
118  // issue when using the GNU C++ STL.
119 
120  unsigned int int_value=(unsigned)h1; // lose data here
121  return ROTL32(int_value, 7)^h2;
122  #else
123  return ROTL64(h1, 7)^h2;
124  #endif
125 }

396  std::size_t h1,
397  std::size_t h2)
398 {
399  const std::size_t h1_tmp=h1;
400 
401  const uint64_t c1=BIG_CONSTANT(0x87c37b91114253d5);
402  const uint64_t c2=BIG_CONSTANT(0x4cf5ad432745937f);
403 
404  h2*=c1;
405  h2=ROTL64(h2,31);
406  h2*=c2;
407 
408  h1^=h2;
409  h1=ROTL64(h1,27);
410  // it may be better to omit the following re-combination according
411  // to the LLBMC benchmark set (reduces the hash collisions in
412  // boolbv_width::get_entry, but slightly increases them elsewhere)
413  h1+=h1_tmp;
414  h1=h1*5+0x52dce729;
415 
416  return h1;
