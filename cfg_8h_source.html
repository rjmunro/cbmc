348 template<class T, typename P, typename I>
349 void cfg_baset<T, P, I>::compute_edges_function_call(
350  const goto_functionst &goto_functions,
351  const goto_programt &goto_program,
352  const goto_programt::instructiont &instruction,
353  goto_programt::const_targett next_PC,
354  entryt &entry)
355 {
356  const exprt &function=
357  to_code_function_call(instruction.code).function();
358 
359  if(function.id()!=ID_symbol)
360  return;
361 
362  const irep_idt &identifier=
363  to_symbol_expr(function).get_identifier();
364 
365  goto_functionst::function_mapt::const_iterator f_it=
366  goto_functions.function_map.find(identifier);
367 
368  if(f_it!=goto_functions.function_map.end() &&
369  f_it->second.body_available())
370  {
371  // get the first instruction
372  goto_programt::const_targett i_it=
373  f_it->second.body.instructions.begin();
374 
375  goto_programt::const_targett e_it=
376  f_it->second.body.instructions.end();
377 
378  goto_programt::const_targett last_it=e_it; last_it--;
379 
380  if(i_it!=e_it)
381  {
382  // nonempty function
383  this->add_edge(entry, entry_map[i_it]);
384 
385  // add the last instruction as predecessor of the return location
386  if(next_PC!=goto_program.instructions.end())
387  this->add_edge(entry_map[last_it], entry_map[next_PC]);
388  }
389  else if(next_PC!=goto_program.instructions.end())
390  {
391  // empty function
392  this->add_edge(entry, entry_map[next_PC]);
393  }
394  }
395  else if(next_PC!=goto_program.instructions.end())
396  this->add_edge(entry, entry_map[next_PC]);
397 }

439 template<class T, typename P, typename I>
440 void cfg_baset<T, P, I>::compute_edges(
441  const goto_functionst &goto_functions,
442  const goto_programt &goto_program,
443  I PC)
444 {
445  const goto_programt::instructiont &instruction=*PC;
446  entryt entry=entry_map[PC];
447  (*this)[entry].PC=PC;
448  goto_programt::const_targett next_PC(PC);
449  next_PC++;
450 
451  // compute forward edges first
452  switch(instruction.type)
453  {
454  case GOTO:
455  compute_edges_goto(goto_program, instruction, next_PC, entry);
456  break;
457 
458  case CATCH:
459  compute_edges_catch(goto_program, instruction, next_PC, entry);
460  break;
461 
462  case THROW:
463  compute_edges_throw(goto_program, instruction, next_PC, entry);
464  break;
465 
466  case START_THREAD:
467  compute_edges_start_thread(
468  goto_program,
469  instruction,
470  next_PC,
471  entry);
472  break;
473 
474  case FUNCTION_CALL:
475  compute_edges_function_call(
476  goto_functions,
477  goto_program,
478  instruction,
479  next_PC,
480  entry);
481  break;
482 
483  case END_THREAD:
484  case END_FUNCTION:
485  // no successor
486  break;
487 
488  case ASSUME:
489  // false guard -> no successor
490  if(instruction.guard.is_false())
491  break;
492 
493  case ASSIGN:
494  case ASSERT:
495  case OTHER:
496  case RETURN:
497  case SKIP:
498  case LOCATION:
499  case ATOMIC_BEGIN:
500  case ATOMIC_END:
501  case DECL:
502  case DEAD:
503  if(next_PC!=goto_program.instructions.end())
504  this->add_edge(entry, entry_map[next_PC]);
505  break;
506 
507  case NO_INSTRUCTION_TYPE:
508  assert(false);
509  break;
510  }
511 }
