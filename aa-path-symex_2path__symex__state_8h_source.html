281  // On first call, O(N) where N is the length of the execution path
282  // leading to this state. Subsequent calls run in constant time.
283  const branchest& get_branches()
284  {
285  if(!branches.empty() || history.is_nil())
286  return branches;
287 
288  // Contrarily, suppose this state is lazy. Since we did not
289  // return above, branches.empty(). By definition of is_lazy(),
290  // branches_restore must have been a negative number. Since
291  // its type is unsigned, however, this is impossible.
292  assert(!is_lazy());
293 
294  path_symex_step_reft step_ref=history;
295  assert(!step_ref.is_nil());
296  loc_reft next_loc_ref=pc();
297  for(;;)
298  {
299  if(step_ref.is_nil())
300  break;
301 
302  const loct &loc=locs[step_ref->pc];
303  if(loc.target->is_goto())
304  {
305  branches.push_back(loc.branch_target==next_loc_ref);
306  assert(branches.back() || step_ref->pc.next_loc()==next_loc_ref);
307  }
308 
309  next_loc_ref=step_ref->pc;
310  step_ref=step_ref->predecessor;
311  }
312 
313  // preserve non-laziness of this state (see first assertion)
314  branches_restore=branches.size();
315 
316  // TODO: add "path_symex_step::branch_depth" field so we can
317  // reserve() the right capacity of vector and avoid reverse().
318  std::reverse(branches.begin(), branches.end());
319 
320  return branches;
321  }
