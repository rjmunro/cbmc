119 template <class P, class T, bool post_dom>
120 void cfg_dominators_templatet<P, T, post_dom>::fixedpoint(P &program)
121 {
122  std::list<T> worklist;
123 
124  if(program.instructions.empty())
125  return;
126 
127  if(post_dom)
128  entry_node = --program.instructions.end();
129  else
130  entry_node = program.instructions.begin();
131  typename cfgt::nodet &n=cfg[cfg.entry_map[entry_node]];
132  n.dominators.insert(entry_node);
133 
134  for(typename cfgt::edgest::const_iterator
135  s_it=(post_dom?n.in:n.out).begin();
136  s_it!=(post_dom?n.in:n.out).end();
137  ++s_it)
138  worklist.push_back(cfg[s_it->first].PC);
139 
140  while(!worklist.empty())
141  {
142  // get node from worklist
143  T current=worklist.front();
144  worklist.pop_front();
145 
146  bool changed=false;
147  typename cfgt::nodet &node=cfg[cfg.entry_map[current]];
148  if(node.dominators.empty())
149  for(typename cfgt::edgest::const_iterator
150  p_it=(post_dom?node.out:node.in).begin();
151  !changed && p_it!=(post_dom?node.out:node.in).end();
152  ++p_it)
153  if(!cfg[p_it->first].dominators.empty())
154  {
155  node.dominators=cfg[p_it->first].dominators;
156  node.dominators.insert(current);
157  changed=true;
158  }
159 
160  // compute intersection of predecessors
161  for(typename cfgt::edgest::const_iterator
162  p_it=(post_dom?node.out:node.in).begin();
163  p_it!=(post_dom?node.out:node.in).end();
164  ++p_it)
165  {
166  const target_sett &other=cfg[p_it->first].dominators;
167  if(other.empty())
168  continue;
169 
170  typename target_sett::const_iterator n_it=node.dominators.begin();
171  typename target_sett::const_iterator o_it=other.begin();
172 
173  // in-place intersection. not safe to use set_intersect
174  while(n_it!=node.dominators.end() && o_it!=other.end())
175  {
176  if(*n_it==current) ++n_it;
177  else if(*n_it<*o_it) { changed=true; node.dominators.erase(n_it++); }
178  else if(*o_it<*n_it) ++o_it;
179  else { ++n_it; ++o_it; }
180  }
181 
182  while(n_it!=node.dominators.end())
183  {
184  if(*n_it==current)
185  ++n_it;
186  else
187  {
188  changed=true;
189  node.dominators.erase(n_it++);
190  }
191  }
192  }
193 
194  if(changed) // fixed point for node reached?
195  {
196  for(typename cfgt::edgest::const_iterator
197  s_it=(post_dom?node.in:node.out).begin();
198  s_it!=(post_dom?node.in:node.out).end();
199  ++s_it)
200  {
201  worklist.push_back(cfg[s_it->first].PC);
202  }
203  }
204  }
205 }
